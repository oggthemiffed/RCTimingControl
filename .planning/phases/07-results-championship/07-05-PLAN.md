```tsx
function LapTimesPanel({ entryId, lapHistory }: { entryId: number; lapHistory: PositionAtLap[] }) {
  // lapHistory is PositionAtLap[] — one record per lap per driver showing position.
  // Individual lap times are NOT in lapHistory; they're not in the current DTO.
  // RESEARCH.md Patterns.md note: "Individual lap times are in ResultRow" — check actual DTO.
  // Since lapHistory only has lapNumber+entryId+position, display what's available:
  const laps = lapHistory.filter(l => l.entryId === entryId);
  if (laps.length === 0) {
    return <p className="text-xs text-muted-foreground py-1">No lap data recorded.</p>;
  }
  return (
    <table className="text-xs font-mono w-full">
      <tbody>
        {laps.map(l => (
          <tr key={l.lapNumber}>
            <td className="pr-4 text-muted-foreground py-0.5">Lap {l.lapNumber}</td>
            <td className="text-right">P{l.position}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
}
```

Note: If `lapHistory` only contains position-at-each-lap (not raw lap times), this panel shows position progression per lap. Actual per-lap times would require a DTO change (a future enhancement). Display what the API provides.
