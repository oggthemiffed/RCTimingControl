```typescript
import { useCallback, useEffect, useRef, useState } from 'react';
import { getRaceClipMap, AudioSettingsDto } from '@/lib/audioApi';

interface UseAnnouncementsOptions {
  raceId: number;
  settings: AudioSettingsDto | null;
  volume?: number;
}

export function useAnnouncements({ raceId, settings, volume = 0.8 }: UseAnnouncementsOptions) {
  const [clips, setClips] = useState<Record<string, string>>({});
  const [isLoading, setIsLoading] = useState(false);
  const audioCache = useRef<Map<string, HTMLAudioElement>>(new Map());

  // Load clips on GRID state
  const loadClips = useCallback(async () => {
    setIsLoading(true);
    try {
      const response = await getRaceClipMap(raceId);
      setClips(response.data);
      // Pre-cache Audio objects
      Object.entries(response.data).forEach(([key, url]) => {
        const audio = new Audio(url);
        audio.volume = volume;
        audioCache.current.set(key, audio);
      });
    } catch (e) {
      console.warn('Failed to load audio clips:', e);
    } finally {
      setIsLoading(false);
    }
  }, [raceId, volume]);

  // Play a clip by key
  const play = useCallback((clipKey: string) => {
    const audio = audioCache.current.get(clipKey);
    if (audio) {
      audio.currentTime = 0;
      audio.play().catch(console.warn);
    } else {
      // Fallback to speech synthesis
      const text = clipKeyToText(clipKey);
      fallbackSpeak(text);
    }
  }, []);

  // Web Speech API fallback (AUDIO-11)
  const fallbackSpeak = useCallback((text: string) => {
    if (!window.speechSynthesis) return;
    const utterance = new SpeechSynthesisUtterance(text);
    window.speechSynthesis.speak(utterance);
  }, []);

  // Play beep sounds (AUDIO-04)
  const playBeep = useCallback((improving: boolean) => {
    if (settings && !settings.announceLapBeep) return;
    
    try {
      const ctx = new AudioContext();
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      gain.gain.value = volume;
      osc.frequency.value = improving ? 880 : 440;
      osc.connect(gain);
      gain.connect(ctx.destination);
      osc.start();
      osc.stop(ctx.currentTime + 0.2);
    } catch (e) {
      console.warn('Beep failed:', e);
    }
  }, [settings, volume]);

  // Convert clip key to fallback text
  const clipKeyToText = (key: string): string => {
    if (key.startsWith('countdown-')) {
      const seconds = parseInt(key.replace('countdown-', ''));
      const minutes = Math.floor(seconds / 60);
      return minutes > 0 ? `${minutes} minute${minutes > 1 ? 's' : ''}` : `${seconds} seconds`;
    }
    if (key.startsWith('car-')) {
      return `Car ${key.replace('car-', '')}`;
    }
    if (key.startsWith('finish-')) {
      return 'Finished';
    }
    return '';
  };

  // Test audio
  const testAudio = useCallback(() => {
    fallbackSpeak('Test. Race one. Positions: first, Car 12. Second, Car 7. Third, Car 3.');
  }, [fallbackSpeak]);

  return {
    clips,
    isLoading,
    loadClips,
    play,
    playBeep,
    fallbackSpeak,
    testAudio,
  };
}
```

**4. Create usePracticeTiming.ts**:
```typescript
import { useCallback, useEffect, useState } from 'react';
import { useStomp } from '@/hooks/useStomp';
import { getSnapshot, PracticeTimingRowDto } from '@/lib/practiceApi';

export function usePracticeTiming(sessionId: number) {
  const [rows, setRows] = useState<PracticeTimingRowDto[]>([]);
  const [unknownTransponders, setUnknownTransponders] = useState<string[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  // Load initial snapshot
  useEffect(() => {
    const load = async () => {
      setIsLoading(true);
      try {
        const response = await getSnapshot(sessionId);
        setRows(response.data);
      } catch (e) {
        console.error('Failed to load snapshot:', e);
      } finally {
        setIsLoading(false);
      }
    };
    load();
  }, [sessionId]);

  // Subscribe to timing updates
  useStomp({
    topic: `/topic/practice/${sessionId}/timing`,
    onMessage: useCallback((data: PracticeTimingRowDto[]) => {
      setRows(data);
    }, []),
  });

  // Subscribe to unknown transponder updates
  useStomp({
    topic: `/topic/practice/${sessionId}/unknown-transponder`,
    onMessage: useCallback((data: string[]) => {
      setUnknownTransponders(data);
    }, []),
  });

  return {
    rows,
    unknownTransponders,
    isLoading,
  };
}
```

```typescript
  const [isOpen, setIsOpen] = useState(false);
  const [settings, setSettings] = useState({
    announceCountdown: true,
    announceStagger: true,
    announceLapBeep: true,
    announceFinish: true,
    announceRunningOrder: true,
    runningOrderDepth: 3,

    ? 'bg-[var(--flag-yellow)]'
```

```typescript
  const id = parseInt(sessionId!, 10);

  });

          <Badge variant={statusBadgeVariant}>{session.status}</Badge>
          <p className="text-xs text-muted-foreground">Best {session.bestLapN} consecutive laps</p>
          {session.eventName && (

          )}
        </div>

              <h3 className="text-lg font-semibold mt-4">Session not started</h3>
              <p className="text-muted-foreground">Press Start to begin timing. Transponders will be detected automatically.</p>
              <Button className="mt-4" onClick={() => startMutation.mutate()}>
                ▶ Start Practice
              </Button>
            </div>

            <>
              <div className="flex items-center justify-between px-4 py-3 border-b">
                <span className="text-xl font-semibold font-mono tabular-nums">

            </>
          )}
        </div>
      </div>
    </RaceControlLayout>
  );
}
```
