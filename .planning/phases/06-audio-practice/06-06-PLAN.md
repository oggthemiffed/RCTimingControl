import { useCallback, useEffect, useRef, useState } from 'react';
import { useStomp } from '@/hooks/useStomp';
import { getRaceClipMap, AudioSettingsDto } from '@/lib/audioApi';

interface UseAnnouncementsOptions {
  raceId: number;
  settings: AudioSettingsDto | null;
  volume?: number;
}

interface RunningOrderEvent {
  type: 'running-order';
  positions: string[];  // ordered list of racer names
}

export function useAnnouncements({ raceId, settings, volume = 0.8 }: UseAnnouncementsOptions) {
  const [clips, setClips] = useState<Record<string, string>>({});
  const [isLoading, setIsLoading] = useState(false);
  const audioCache = useRef<Map<string, HTMLAudioElement>>(new Map());

  // Load clips on GRID state
  const loadClips = useCallback(async () => {
    setIsLoading(true);
    try {
      const response = await getRaceClipMap(raceId);
      setClips(response.data);
      // Pre-cache Audio objects
      Object.entries(response.data).forEach(([key, url]) => {
        const audio = new Audio(url);
        audio.volume = volume;
        audioCache.current.set(key, audio);
      });
    } catch (e) {
      console.warn('Failed to load audio clips:', e);
    } finally {
      setIsLoading(false);
    }
  }, [raceId, volume]);

  // Play a clip by key
  const play = useCallback((clipKey: string) => {
    const audio = audioCache.current.get(clipKey);
    if (audio) {
      audio.currentTime = 0;
      audio.play().catch(console.warn);
    } else {
      // Fallback to speech synthesis
      const text = clipKeyToText(clipKey);
      fallbackSpeak(text);
    }
  }, []);

  // Web Speech API fallback (AUDIO-11)
  const fallbackSpeak = useCallback((text: string) => {
    if (!window.speechSynthesis) return;
    const utterance = new SpeechSynthesisUtterance(text);
    window.speechSynthesis.speak(utterance);
  }, []);

  // Synthesize running order announcement (AUDIO-06)
  const synthesizeRunningOrder = useCallback((positions: string[]) => {
    if (!window.speechSynthesis || !settings?.announceRunningOrder) return;
    const text = `Running order: ${positions.map((name, i) => `${i + 1}, ${name}`).join('. ')}.`;
    const utterance = new SpeechSynthesisUtterance(text);
    window.speechSynthesis.speak(utterance);
  }, [settings]);

  // Subscribe to running order audio events (AUDIO-06)
  useStomp({
    topic: `/topic/race/${raceId}/audio`,
    onMessage: useCallback((data: RunningOrderEvent) => {
      if (data.type === 'running-order' && data.positions?.length > 0) {
        synthesizeRunningOrder(data.positions);
      }
    }, [synthesizeRunningOrder]),
  });

  // Play beep sounds (AUDIO-04)
  const playBeep = useCallback((improving: boolean) => {
    if (settings && !settings.announceLapBeep) return;
    
    try {
      const ctx = new AudioContext();
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      gain.gain.value = volume;
      osc.frequency.value = improving ? 880 : 440;
      osc.connect(gain);
      gain.connect(ctx.destination);
      osc.start();
      osc.stop(ctx.currentTime + 0.2);
    } catch (e) {
      console.warn('Beep failed:', e);
    }
  }, [settings, volume]);

  // Convert clip key to fallback text
  const clipKeyToText = (key: string): string => {
    if (key.startsWith('countdown-')) {
      const seconds = parseInt(key.replace('countdown-', ''));
      const minutes = Math.floor(seconds / 60);
      return minutes > 0 ? `${minutes} minute${minutes > 1 ? 's' : ''}` : `${seconds} seconds`;
    }
    if (key.startsWith('car-')) {
      return `Car ${key.replace('car-', '')}`;
    }
    if (key.startsWith('finish-')) {
      return 'Finished';
    }
    return '';
  };

  // Test audio
  const testAudio = useCallback(() => {
    fallbackSpeak('Test. Race one. Positions: first, Car 12. Second, Car 7. Third, Car 3.');
  }, [fallbackSpeak]);

  return {
    clips,
    isLoading,
    loadClips,
    play,
    playBeep,
    fallbackSpeak,
    synthesizeRunningOrder,
    testAudio,
  };
}

import { useCallback, useEffect, useState } from 'react';
import { useStomp } from '@/hooks/useStomp';
import { getSnapshot, PracticeTimingRowDto } from '@/lib/practiceApi';

export function usePracticeTiming(sessionId: number) {
  const [rows, setRows] = useState<PracticeTimingRowDto[]>([]);
  const [unknownTransponders, setUnknownTransponders] = useState<string[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  // Load initial snapshot
  useEffect(() => {
    const load = async () => {
      setIsLoading(true);
      try {
        const response = await getSnapshot(sessionId);
        setRows(response.data);
      } catch (e) {
        console.error('Failed to load snapshot:', e);
      } finally {
        setIsLoading(false);
      }
    };
    load();
  }, [sessionId]);

  // Subscribe to timing updates
  useStomp({
    topic: `/topic/practice/${sessionId}/timing`,
    onMessage: useCallback((data: PracticeTimingRowDto[]) => {
      setRows(data);
    }, []),
  });

  // Subscribe to unknown transponder updates
  useStomp({
    topic: `/topic/practice/${sessionId}/unknown-transponder`,
    onMessage: useCallback((data: string[]) => {
      setUnknownTransponders(data);
    }, []),
  });

  return {
    rows,
    unknownTransponders,
    isLoading,
  };
}

// API calls to /api/v1/race-control/settings/audio (Issue 4: settings must persist)
const getAudioSettings = () => 
  fetch('/api/v1/race-control/settings/audio', { credentials: 'include' }).then(r => r.json());

const patchAudioSettings = (settings: AudioSettingsDto) =>
  fetch('/api/v1/race-control/settings/audio', {
    method: 'PATCH',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(settings),
  }).then(r => r.json());

interface AudioSettingsDto {
  announceCountdown: boolean;
  announceStagger: boolean;
  announceLapBeep: boolean;
  announceFinish: boolean;
  announceRunningOrder: boolean;
  runningOrderDepth: number;
}

interface AudioSettingsPanelProps {
  raceId: number;
}

export function AudioSettingsPanel({ raceId }: AudioSettingsPanelProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [volume, setVolume] = useState(() => {
    const stored = localStorage.getItem('rc-audio-volume');
    return stored ? parseInt(stored) : 80;
  });
  const queryClient = useQueryClient();

  // Fetch settings from API (persisted per club)
  const { data: settings, isLoading } = useQuery({
    queryKey: ['audio-settings'],
    queryFn: getAudioSettings,
  });

  // Mutation to save settings
  const mutation = useMutation({
    mutationFn: patchAudioSettings,
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['audio-settings'] }),
  });
  
  const { testAudio } = useAnnouncements({ raceId, settings: settings || null, volume: volume / 100 });

  // Persist volume to localStorage
  useEffect(() => {
    localStorage.setItem('rc-audio-volume', String(volume));
  }, [volume]);

  const toggleSetting = (key: keyof AudioSettingsDto) => {
    if (!settings) return;
    const updated = { ...settings, [key]: !settings[key] };
    mutation.mutate(updated);
  };

  const updateDepth = (depth: number) => {
    if (!settings) return;
    mutation.mutate({ ...settings, runningOrderDepth: depth });

  const allEnabled = settings && [
    settings.announceCountdown,
    settings.announceStagger,
    settings.announceLapBeep,
    settings.announceFinish,
    settings.announceRunningOrder,

    settings.announceStagger,
    settings.announceLapBeep,
    settings.announceFinish,
    settings.announceRunningOrder,

  const statusDot = allEnabled

        <Volume2 className="h-4 w-4" />
        <span className="text-sm font-normal">Audio Settings</span>
        <span className={`ml-auto h-2 w-2 rounded-full ${statusDot}`} />
        {isOpen ? <ChevronUp className="h-4 w-4" /> : <ChevronDown className="h-4 w-4" />}
      </button>

      {/* Expanded panel */}
      {isOpen && (
        <div className="px-4 pb-4 space-y-4">
          {isLoading ? (
            <div className="flex items-center gap-2 py-4">
              <Loader2 className="h-4 w-4 animate-spin" />

              <p className="text-xs font-semibold text-muted-foreground">Announcement Types</p>

import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { Input } from '@/components/ui/input';
import { Button } from '@/components/ui/button';

interface PhoneticDto {
  displayName: string;
  phoneticName: string | null;
}

// API functions
const getRacerPhonetic = (userId: number): Promise<PhoneticDto> =>
  fetch(`/api/v1/admin/audio/racer/${userId}/phonetic`, { credentials: 'include' }).then(r => r.json());

const updateRacerPhonetic = (userId: number, phoneticName: string | null): Promise<PhoneticDto> =>
  fetch(`/api/v1/admin/audio/racer/${userId}/phonetic`, {
    method: 'PUT',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ phoneticName }),
  }).then(r => r.json());

const regenerateNameClip = (userId: number): Promise<void> =>
  fetch(`/api/v1/admin/audio/racer/${userId}/name-clip`, {
    method: 'DELETE',
    credentials: 'include',
  }).then(() => undefined);

// Component (add to existing racer detail page or create new):
function PhoneticNameSection({ userId }: { userId: number }) {
  const queryClient = useQueryClient();
  const [editedPhonetic, setEditedPhonetic] = useState<string>('');
  
  const { data: phonetic, isLoading } = useQuery({
    queryKey: ['racer-phonetic', userId],
    queryFn: () => getRacerPhonetic(userId),
    onSuccess: (data) => setEditedPhonetic(data.phoneticName || ''),
  });

  const saveMutation = useMutation({
    mutationFn: (phoneticName: string | null) => updateRacerPhonetic(userId, phoneticName),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['racer-phonetic', userId] });
      toast.success('Phonetic name saved.');
    },
  });

  const clearAndRegenerateMutation = useMutation({
    mutationFn: async () => {
      await updateRacerPhonetic(userId, null);
      await regenerateNameClip(userId);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['racer-phonetic', userId] });
      setEditedPhonetic('');
      toast.success('Phonetic name cleared. Clip will regenerate.');
    },
  });

  if (isLoading || !phonetic) return <div>Loading...</div>;
