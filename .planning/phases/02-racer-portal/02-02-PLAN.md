    ```java
    package dev.monkeypatch.rctiming.query.car;

    import dev.monkeypatch.rctiming.jooq.generated.tables.Cars;
    import dev.monkeypatch.rctiming.jooq.generated.tables.CarTagCategories;
    import dev.monkeypatch.rctiming.jooq.generated.tables.CarTagValues;
    import org.jooq.DSLContext;
    import org.jooq.Record2;
    import org.springframework.stereotype.Service;
    import org.springframework.transaction.annotation.Transactional;

    import java.util.*;
    import java.util.stream.Collectors;

    @Service
    @Transactional(readOnly = true)
    public class CarQueryService {

        private static final Cars CARS = Cars.CARS;
        private static final CarTagCategories CTC = CarTagCategories.CAR_TAG_CATEGORIES;
        private static final CarTagValues CTV = CarTagValues.CAR_TAG_VALUES;

        private final DSLContext dsl;

        public CarQueryService(DSLContext dsl) {
            this.dsl = dsl;
        }

        public List&lt;CarWithTagsDto&gt; getActiveCarsForUser(Long userId) {
            // 1. Load active cars for user
            var carRows = dsl.selectFrom(CARS)
                    .where(CARS.USER_ID.eq(userId).and(CARS.ARCHIVED.isFalse()))
                    .orderBy(CARS.CREATED_AT.asc())
                    .fetch();
            if (carRows.isEmpty()) return List.of();

            List&lt;Long&gt; carIds = carRows.map(r -&gt; r.get(CARS.ID));

            // 2. Load tag values for those cars, joining category name
            Map&lt;Long, Map&lt;String, String&gt;&gt; tagsByCarId = dsl
                    .select(CTV.CAR_ID, CTC.NAME, CTV.VALUE)
                    .from(CTV)
                    .join(CTC).on(CTC.ID.eq(CTV.CATEGORY_ID))
                    .where(CTV.CAR_ID.in(carIds))
                    .fetchGroups(CTV.CAR_ID).entrySet().stream()
                    .collect(Collectors.toMap(
                            Map.Entry::getKey,
                            e -&gt; e.getValue().stream().collect(Collectors.toMap(
                                    r -&gt; r.get(CTC.NAME),
                                    r -&gt; r.get(CTV.VALUE),
                                    (a, b) -&gt; a, LinkedHashMap::new))
                    ));

            // 3. Assemble DTOs preserving car order
            return carRows.stream()
                    .map(r -&gt; new CarWithTagsDto(
                            r.get(CARS.ID),
                            r.get(CARS.USER_ID),
                            r.get(CARS.NAME),
                            r.get(CARS.PRIMARY_CLASS_ID),
                            r.get(CARS.NOTES),
                            r.get(CARS.ARCHIVED),
                            tagsByCarId.getOrDefault(r.get(CARS.ID), Map.of())
                    ))
                    .toList();
        }

        public Optional&lt;CarWithTagsDto&gt; getCarForUser(Long carId, Long userId) {
            var row = dsl.selectFrom(CARS)
                    .where(CARS.ID.eq(carId).and(CARS.USER_ID.eq(userId)))
                    .fetchOptional();
            if (row.isEmpty()) return Optional.empty();

            Map&lt;String, String&gt; tags = dsl.select(CTC.NAME, CTV.VALUE)
                    .from(CTV)
                    .join(CTC).on(CTC.ID.eq(CTV.CATEGORY_ID))
                    .where(CTV.CAR_ID.eq(carId))
                    .fetchMap(CTC.NAME, CTV.VALUE);

            var r = row.get();
            return Optional.of(new CarWithTagsDto(
                    r.get(CARS.ID), r.get(CARS.USER_ID), r.get(CARS.NAME),
                    r.get(CARS.PRIMARY_CLASS_ID), r.get(CARS.NOTES), r.get(CARS.ARCHIVED),
                    tags
            ));
        }
    }
    ```

    Critical rules:
    - `@Transactional(readOnly = true)` at class level — never without this annotation (Anti-Patterns section).
    - DSLContext constructor injection — Spring Boot auto-configures the bean from the JPA datasource.
    - Use generated table constants from `dev.monkeypatch.rctiming.jooq.generated.tables.*` — do NOT write raw SQL strings.
    - Archived cars are filtered at the query level (RACER-06): `CARS.ARCHIVED.isFalse()`.
    - Ownership filter is part of the query (`CARS.USER_ID.eq(userId)`) — never loaded then filtered in Java.
    - Two round-trips to DB (cars, then tags) is acceptable and avoids Cartesian explosion. Do NOT use `leftJoin` to tags in the same query — it creates duplicate car rows.
