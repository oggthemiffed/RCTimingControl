    ```tsx
    const [polling, setPolling] = useState(false);
    const [attempts, setAttempts] = useState(0);
    const [testResult, setTestResult] = useState<'idle' | 'connected' | 'timeout'>('idle');

    const statusQuery = useQuery({

      enabled: polling && attempts < MAX_ATTEMPTS,
      refetchInterval: polling && attempts < MAX_ATTEMPTS ? 2000 : false,
      staleTime: 0,
    });

    useEffect(() => {
      if (!polling) return;
      if (statusQuery.data?.forwarderState === 'CONNECTED') {
        setTestResult('connected');
        setPolling(false);
      } else if (statusQuery.isFetched) {
        setAttempts(a => {
          const next = a + 1;
          if (next >= MAX_ATTEMPTS) {
            setTestResult('timeout');
            setPolling(false);
          }
          return next;
        });
      }
    }, [statusQuery.data, statusQuery.isFetched, polling]);

    const onTestConnection = () => {
      setAttempts(0);
      setTestResult('idle');
      setPolling(true);
    };
    ```
