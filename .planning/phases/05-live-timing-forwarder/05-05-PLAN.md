    ```tsx
    import { Badge } from '@/components/ui/badge';
    import { useStomp } from '@/hooks/race-control/useStomp';
    import type { ConnectionState, ForwarderStatusDto } from '@/types/forwarder';

    function stateColor(state: ConnectionState | undefined): string {
      switch (state) {
        case 'CONNECTED':    return 'var(--flag-green)';
        case 'RECONNECTING': return 'var(--flag-yellow)';
        case 'DISCONNECTED': return 'var(--flag-red)';
        default:             return 'var(--muted-foreground)';
      }
    }

    function stateLabel(state: ConnectionState | undefined, prefix: string): string {
      switch (state) {
        case 'CONNECTED':    return `${prefix} connected`;
        case 'RECONNECTING': return `${prefix} reconnecting…`;
        case 'DISCONNECTED': return `${prefix} disconnected`;
        default:             return `${prefix} —`;
      }
    }

    interface StatusPillProps {
      state: ConnectionState | undefined;
      prefix: string;
    }

    function StatusPill({ state, prefix }: StatusPillProps) {
      const color = stateColor(state);
      const label = stateLabel(state, prefix);
      const isConnected = state === 'CONNECTED';
      return (
        <Badge
          variant="outline"
          className="gap-2 h-6 px-3"
          aria-label={`${prefix} connection status: ${state ?? 'unknown'}`}
        >
          <span
            aria-hidden="true"
            className={`inline-block h-2 w-2 rounded-full${isConnected ? ' animate-pulse' : ''}`}
            style={{ backgroundColor: color }}
          />
          <span className="text-xs font-normal" style={{ color }}>
            {label}
          </span>
        </Badge>
      );
    }

    export function ForwarderStatusBar() {
      const { data: status } = useStomp<ForwarderStatusDto>(
        '/topic/system/forwarder-status'
      );

      return (
        <div className="flex h-8 items-center gap-3 px-4 bg-card border-b shrink-0">
          <StatusPill state={status?.decoderState} prefix="DECODER" />
          <StatusPill state={status?.forwarderState} prefix="FORWARDER" />
        </div>
      );
    }
    ```

    ```tsx
    import { useState } from 'react';
    import {
      Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter
    } from '@/components/ui/dialog';
    import {
      Select, SelectContent, SelectItem, SelectTrigger, SelectValue
    } from '@/components/ui/select';
    import { Button } from '@/components/ui/button';
    import { Input } from '@/components/ui/input';
    import { toast } from 'sonner';
    import { linkUnknownTransponder } from '@/lib/raceControlApi';

    // ... props interface above ...

    export function UnknownTransponderLinkDialog({
      open, onOpenChange, raceId, transponderNumber, passingCount, entries
    }: UnknownTransponderLinkDialogProps) {
      const [selectedEntryId, setSelectedEntryId] = useState<string>('');
      const [loading, setLoading] = useState(false);

      async function handleLink() {
        if (!selectedEntryId) return;
        setLoading(true);
        try {
          const result = await linkUnknownTransponder(raceId, transponderNumber, Number(selectedEntryId));
          onOpenChange(false);
          toast.success(`Transponder linked. ${result.lapsCredited} lap(s) credited to ${result.racerName}.`);

          const msg = (err as { response?: { data?: { message?: string } } })
            ?.response?.data?.message ?? 'Please try again.';
          toast.error(`Failed to link transponder. ${msg}`);
        } finally {
          setLoading(false);
        }
      }

      return (
        <Dialog open={open} onOpenChange={onOpenChange}>
          <DialogContent className="sm:max-w-md">
            <DialogHeader>
              <DialogTitle>Link Unknown Transponder</DialogTitle>
            </DialogHeader>

            <div className="space-y-4">
              <div className="space-y-2">
                <label className="text-sm font-normal">Transponder ID</label>
                <Input readOnly value={transponderNumber} className="font-mono text-sm" />
              </div>

              <p className="text-xs text-muted-foreground">
                This transponder has {passingCount} passing(s) since race start.
                All will be credited to the linked entry immediately.
              </p>

              <div className="space-y-2">
                <label className="text-sm font-normal">Link to entry</label>
                <Select value={selectedEntryId} onValueChange={setSelectedEntryId}>
                  <SelectTrigger>
                    <SelectValue placeholder="Select entry…" />
                  </SelectTrigger>
                  <SelectContent>
                    {entries.map((e) => (
                      <SelectItem key={e.entryId} value={String(e.entryId)}>
                        Car {e.carNumber ?? 'No car number'} — {e.racerName}
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
            </div>

            <DialogFooter>
              <Button variant="ghost" onClick={() => onOpenChange(false)}>
                Keep monitoring
              </Button>
              <Button
                variant="default"
                disabled={!selectedEntryId || loading}
                onClick={handleLink}
              >
                {loading ? 'Linking…' : 'Link Entry'}
              </Button>
            </DialogFooter>
          </DialogContent>
        </Dialog>
      );
    }
    ```

    **B. frontend/src/pages/admin/forwarder/ForwarderTokenPage.tsx** — New page. Create the `frontend/src/pages/admin/forwarder/` directory. Use `useQuery` from TanStack Query v5 (`@tanstack/react-query`) and `useMutation` for mutations — these are already installed in the project (check package.json if unsure, but they are in the stack per CLAUDE.md).

    State machine (four UI states from UI-SPEC Surface 3):
    - `'none'` — no token ever generated
    - `'active'` — token exists and is ACTIVE
    - `'revealed'` — token was just generated/regenerated (one-time reveal)
    - `'revoked'` — token is REVOKED

    Inline confirmation state for Regenerate/Revoke — use a local `confirmAction: 'regenerate' | 'revoke' | null` state variable.

    Full component (implement all four states and inline confirmation):
    ```tsx
    import { useState } from 'react';
    import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
    import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
    import { Button } from '@/components/ui/button';
    import { Input } from '@/components/ui/input';
    import { Badge } from '@/components/ui/badge';
    import { toast } from 'sonner';
    import {
      getForwarderTokenStatus,
      generateForwarderToken,
      revokeForwarderToken,
    } from '@/lib/adminApi';

    export function ForwarderTokenPage() {
      const queryClient = useQueryClient();
      const [revealedToken, setRevealedToken] = useState<string | null>(null);
      const [confirmAction, setConfirmAction] = useState<'regenerate' | 'revoke' | null>(null);

      const { data: tokenStatus, isLoading } = useQuery({
        queryKey: ['forwarder-token-status'],
        queryFn: getForwarderTokenStatus,
      });

      const generateMutation = useMutation({
        mutationFn: generateForwarderToken,
        onSuccess: (data) => {
          setRevealedToken(data.token);
          setConfirmAction(null);
          queryClient.invalidateQueries({ queryKey: ['forwarder-token-status'] });
        },
      });

      const revokeMutation = useMutation({
        mutationFn: revokeForwarderToken,
        onSuccess: () => {
          setRevealedToken(null);
          setConfirmAction(null);
          queryClient.invalidateQueries({ queryKey: ['forwarder-token-status'] });
        },
      });

      function handleCopy() {
        if (!revealedToken) return;
        navigator.clipboard.writeText(revealedToken);
        toast.success('Token copied to clipboard.');
      }

      if (isLoading) return <div className="p-6">Loading…</div>;

      const status = tokenStatus?.status ?? 'NONE';
      const generatedAt = tokenStatus?.generatedAt
        ? new Date(tokenStatus.generatedAt).toLocaleString()
        : null;
      const revokedAt = tokenStatus?.revokedAt
        ? new Date(tokenStatus.revokedAt).toLocaleString()
        : null;

      return (
        <div className="max-w-2xl mx-auto p-6">
          <h1 className="text-base font-semibold mb-6">Forwarder Token</h1>

          <Card>
            <CardHeader>
              <CardTitle className="text-base font-semibold">Forwarder API Token</CardTitle>
            </CardHeader>
            <CardContent className="space-y-4">

              {/* State A — No token */}
              {status === 'NONE' && !revealedToken && (
                <>
                  <p className="text-sm">
                    No token has been generated yet. Generate a token to allow the forwarder
                    application to connect to this service.
                  </p>
                  <Button variant="default" onClick={() => generateMutation.mutate()}>
                    Generate Token
                  </Button>
                </>
              )}

              {/* State C — One-time reveal */}
              {revealedToken && (
                <>
                  <p className="text-sm font-semibold text-destructive">
                    Copy this token now. It will not be shown again.
                  </p>
                  <div className="flex gap-2 items-center">
                    <Input
                      readOnly
                      value={revealedToken}
                      className="font-mono text-sm flex-1"
                      aria-label="Copy token to clipboard"
                    />
                    <Button variant="outline" size="sm" onClick={handleCopy} aria-label="Copy token to clipboard">
                      Copy
                    </Button>
                  </div>
                  <p className="text-xs text-muted-foreground">
                    Paste this value into forwarder.properties as{' '}
                    <code className="font-mono">forwarder.api-token=&lt;value&gt;</code>
                  </p>
                  <Button variant="ghost" onClick={() => setRevealedToken(null)}>
    ```
