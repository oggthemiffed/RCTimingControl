    ```ts
    export const PROXIMITY_CLOSING_DELTA_MS = 500;
    export const PROXIMITY_ABSOLUTE_MAX_MS = 3000;

    export type LiveRacePositionDto = {
      entryId: number;
      position: number;
      lapsCompleted: number;
      lastPassingTimeMs: number;
      bestLapMs: number | null;
      gapToLeaderMs: number | null;
      gapToAheadMs: number | null;
    };

    /** OFFICIAL-01: entryIds whose gapToAheadMs has shrunk by >= PROXIMITY_CLOSING_DELTA_MS since previous. */
    export function computeProximityAlerts(
      current: LiveRacePositionDto[],
      previous: LiveRacePositionDto[] | null,
    ): Set<number> { … }

    /** OFFICIAL-02: entryIds whose lapsCompleted < leaderLaps. */
    export function computeBackmarkers(current: LiveRacePositionDto[]): Set<number> { … }
    ```
    Algorithm details:
    - `computeProximityAlerts`: If `previous` is null or empty, return `new Set<number>()`. For each entry `c` in `current`, find the matching `p` in `previous` by `entryId`. If both have finite `gapToAheadMs` values (not null, not 0), `p.gapToAheadMs > c.gapToAheadMs + PROXIMITY_CLOSING_DELTA_MS`, AND `c.gapToAheadMs <= PROXIMITY_ABSOLUTE_MAX_MS`, include `c.entryId` in the result set.
    - `computeBackmarkers`: Compute `leaderLaps = max(current.map(c => c.lapsCompleted))`. Return a `Set` of every `entryId` whose `lapsCompleted < leaderLaps`. Leader (and any driver on the lead lap) is excluded.
